API_BASE_URL = "https://rickandmortyapi.com/api/character"


def fetch_all_characters() -> List[Dict]:
    """Fetch all characters with pagination support."""
    all_characters = []
    url = API_BASE_URL
    
    # PAGINATION LOOP - ×¢×•×‘×¨ ×¢×œ ×›×œ ×”×¢××•×“×™×
    while url:  # ×›×œ ×¢×•×“ ×™×© URL (×œ× None)
        response = requests.get(url)  # ×©×•×œ×— GET request
        response.raise_for_status()   # ×–×•×¨×§ Exception ×× ×©×’×™××”
        data = response.json()        # ×××™×¨ JSON ×œ-dict
        
        all_characters.extend(data['results'])  # ××•×¡×™×£ ×“××•×™×•×ª
        url = data['info']['next']  # URL ×”×‘× (××• None)
    
    return all_characters


def filter_characters(characters: List[Dict]) -> List[Dict]:
    """Filter by: Human, Alive, from Earth"""
    filtered = []
    
    for char in characters:
        is_human = char['species'] == 'Human'
        is_alive = char['status'] == 'Alive'
        is_from_earth = 'Earth' in char['origin']['name']
        # â†‘ ×œ××” 'in'? ×›×™ ×™×© ×•×¨×™××¦×™×•×ª: "Earth (C-137)"
        
        if is_human and is_alive and is_from_earth:
            filtered.append({
                'Name': char['name'],
                'Location': char['location']['name'],
                'Image': char['image']
            })
    
    return filtered


def write_to_csv(characters: List[Dict], filename: str = 'output.csv'):
    """Write to CSV file."""
    with open(filename, 'w', newline='', encoding='utf-8') as f:
        # newline='' = ××•× ×¢ ×©×•×¨×•×ª ×¨×™×§×•×ª ×‘-Windows
        # encoding='utf-8' = ×ª×•××š ×‘×¢×‘×¨×™×ª
        
        writer = csv.DictWriter(f, fieldnames=['Name', 'Location', 'Image'])
        writer.writeheader()
        writer.writerows(characters)
    
    print(f"Written {len(characters)} characters to {filename}")


def main():
    print("Fetching characters from Rick and Morty API...")
    
    all_characters = fetch_all_characters()
    print(f"Total characters: {len(all_characters)}")  # 826
    
    filtered = filter_characters(all_characters)
    print(f"Matching criteria: {len(filtered)}")  # ~109
    
    write_to_csv(filtered)


if __name__ == "__main__":
    main()

from flask import Flask, jsonify
import requests

app = Flask(__name__)

API_BASE_URL = "https://rickandmortyapi.com/api/character"


def fetch_filtered_characters():
    """Fetch and filter characters."""
    all_characters = []
    url = API_BASE_URL
    
    while url:
        response = requests.get(url)
        data = response.json()
        all_characters.extend(data['results'])
        url = data['info']['next']
    
    # ×¡×™× ×•×Ÿ
    filtered = []
    for char in all_characters:
        if (char['species'] == 'Human' and 
            char['status'] == 'Alive' and 
            'Earth' in char['origin']['name']):
            filtered.append({
                'name': char['name'],
                'location': char['location']['name'],
                'image': char['image']
            })
    
    return filtered


@app.route('/healthcheck', methods=['GET'])
@app.route('/health', methods=['GET'])
def healthcheck():
    """Health check endpoint."""
    return jsonify({
        'status': 'healthy',
        'service': 'rick-and-morty-api'
    }), 200


@app.route('/characters', methods=['GET'])
def get_characters():
    """Get all filtered characters."""
    try:
        characters = fetch_filtered_characters()
        return jsonify({
            'count': len(characters),
            'characters': characters
        }), 200
    except Exception as e:
        return jsonify({'error': str(e)}), 500

def index():
